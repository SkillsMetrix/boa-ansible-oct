from abc import ABC,abstractmethod

class RBI(ABC):
    @abstractmethod
    def withdraw(amount):
        pass
    @abstractmethod
    def deposit(amount):
        pass
    @abstractmethod
    def checkBalance(self):
        pass

from RBIBank import RBI

class SBI(RBI):
    def openFD(self):
        print('SBI Bank open FD Account')
    def checkBalance(self):
        print('checking balance in SBI')
    def withdraw(self,amount):
        print(f"amount withdrawn {self.balance}")
    
    def deposit(self,amount):
        print(f"amount deposit {self.balance}")
