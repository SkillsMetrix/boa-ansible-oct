
- hosts: web
  become: true
  pre_tasks:
    - name: update cache
      apt:
        update_cache: true
  tasks:
    - name: Install required softwares
      apt:
        name:
          - python3
          - python3-pip
          - python3-venv
        state: latest

    - name: create fastAPI Directory on remote machine
      file:
        path: /opt/fastapi_app
        state: directory
        mode: '0755'
    
    - name: copy local directory to target machine
      copy:
        src: userapp/
        dest: /opt/fastapi_app
        mode: '0644'

    - name: create venv
      command: python3 -m venv /opt/fastapi_app/venv
    
    - name: install fastapi and uvicorn 
      command: /opt/fastapi_app/venv/bin/pip install fastapi uvicorn
    
    - name: run the application
      shell: |
        nohup /opt/fastapi_app/venv/bin/uvicorn main:app --host 0.0.0.0 --port 8000 &
      args:
        chdir: /opt/fastapi_app
