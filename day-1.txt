
# class declaration
class DemoClass:
    name='shashi'
    email='shashi@mail.com'

    # constructor declarations

    def __init__(self,city,pincode):
        self.city=city
        self.pincode=pincode

    # declare functions
    def showData(self):
        print(self.name,self.email,self.city,self.pincode)


#object creation

demo= DemoClass('pune',412345)
demo.showData()
